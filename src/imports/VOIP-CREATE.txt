        Log.d(TAG, "SipService created")

        initializeSipManager()
        registerSipAccount()
    }

    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        Log.d(TAG, "SipService started with intent: $intent")
        
        when (intent?.action) {
            ACTION_REGISTER -> registerSipAccount()
            ACTION_UNREGISTER -> unregisterSipAccount()
            ACTION_MAKE_CALL -> {
                val number = intent.getStringExtra(EXTRA_CALL_NUMBER)
                number?.let { makeCall(it) }
            }
            ACTION_ANSWER -> answerCall()
            ACTION_REJECT -> rejectCall()
            ACTION_HANGUP -> hangup()
            ACTION_HOLD -> holdCall()
            ACTION_RESUME -> resumeCall()
            ACTION_MUTE -> muteCall()
            ACTION_UNMUTE -> unmuteCall()
            ACTION_SPEAKER_ON -> enableSpeaker()
            ACTION_SPEAKER_OFF -> disableSpeaker()
            ACTION_TRANSFER -> {
                val number = intent.getStringExtra(EXTRA_TRANSFER_NUMBER)
                number?.let { transferCall(it) }
            }
            ACTION_START_CONFERENCE -> startConference()
        }

        startForeground(NOTIFICATION_ID_SERVICE, createServiceNotification())
        return START_STICKY
    }

    override fun onBind(intent: Intent?): IBinder? = null

    override fun onDestroy() {
        super.onDestroy()
        Log.d(TAG, "SipService destroyed")
        
        scope.cancel()
        reconnectJob?.cancel()
        
        sipAudioCall?.close()
        sipProfile?.let {
            try {
                sipManager?.close(it.uriString)
            } catch (e: Exception) {
                Log.e(TAG, "Error closing SIP profile", e)
            }
        }
        
        releaseWakeLock()
    }

    private fun initializeSipManager() {
        audioManager = getSystemService(Context.AUDIO_SERVICE) as AudioManager
        
        if (SipManager.isVoipSupported(this)) {
            sipManager = SipManager.newInstance(this)
            Log.d(TAG, "SIP Manager initialized successfully")
        } else {
            Log.e(TAG, "SIP is not supported on this device")
            _registrationState.value = RegistrationState.Error("SIP not supported")
        }
    }

    private fun registerSipAccount() {
        scope.launch {
            try {
                _registrationState.value = RegistrationState.Registering
                
                val prefs = getSharedPreferences("sip_prefs", Context.MODE_PRIVATE)
                val username = prefs.getString("username", "100") ?: "100"
                val domain = prefs.getString("domain", "192.168.1.1") ?: "192.168.1.1"
                val password = prefs.getString("password", "password") ?: "password"
                val port = prefs.getInt("port", 5060)
                
                val builder = SipProfile.Builder(username, domain)
                builder.setPort(port)
                builder.setPassword(password)
                builder.setAuthUserName(username)
                builder.setDisplayName("VOIP Mother User")
                builder.setAutoRegistration(true)
                builder.setOutboundProxy("$domain:$port")
                builder.setTransportType(SipProfile.TRANSPORT_UDP)
                
                sipProfile = builder.build()
                
                val pendingIntent = PendingIntent.getBroadcast(
                    this@SipService,
                    0,
                    Intent(ACTION_SIP_CALL),
                    PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
                )
                
                sipManager?.open(
                    sipProfile,
                    pendingIntent,
                    object : SipRegistrationListener {
                        override fun onRegistering() {
                            Log.d(TAG, "Registering to SIP server...")
                            _registrationState.value = RegistrationState.Registering
                        }

                        override fun onRegistrationDone(sipProfileUri: String?, expiry: Long) {
                            Log.d(TAG, "Registration successful: $sipProfileUri")
                            _registrationState.value = RegistrationState.Registered(sipProfileUri ?: "")
                            startReconnectWatcher()
                        }

                        override fun onRegistrationFailed(
                            sipProfileUri: String?,
                            errorCode: Int,
                            errorMessage: String?
                        ) {
                            Log.e(TAG, "Registration failed: $errorMessage (code: $errorCode)")
                            _registrationState.value = RegistrationState.RegistrationFailed(
                                errorCode,
                                errorMessage ?: "Unknown error"
                            )
                            scheduleReconnect()
                        }
                    }
                )
                
            } catch (e: Exception) {
                Log.e(TAG, "Error registering SIP account", e)
                _registrationState.value = RegistrationState.Error(e.message ?: "Unknown error")
            }
        }
    }

    private fun unregisterSipAccount() {
        scope.launch {
            try {
                sipProfile?.let {
                    sipManager?.close(it.uriString)
                }
                _registrationState.value = RegistrationState.Unregistered
            } catch (e: Exception) {
                Log.e(TAG, "Error unregistering", e)
            }
        }
    }

    private fun makeCall(number: String) {
        scope.launch {
            try {
                if (sipProfile == null) {
                    Log.e(TAG, "SIP profile is null")
                    return@launch
                }

                val callIntent = Intent(this@SipService, SipService::class.java).apply {
                    action = ACTION_SIP_CALL
                }
                val pendingIntent = PendingIntent.getBroadcast(
                    this@SipService,
                    0,
                    callIntent,
                    PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
                )

                sipAudioCall = sipManager?.makeAudioCall(
                    sipProfile,
                    SipProfile.Builder(number, getDomain()).build(),
                    pendingIntent,
                    30
                )

                if (sipAudioCall != null) {
                    _currentCall.value = CallInfo(
                        callId = System.currentTimeMillis().toString(),
                        number = number,
                        direction = CallDirection.OUTGOING,
                        state = CallState.CONNECTING,
                        startTime = System.currentTimeMillis()
                    )
                    
                    setupCallListener()
                    showCallNotification(number, CallState.CONNECTING)
                } else {
                    Log.e(TAG, "Failed to make call")
                }
            } catch (e: Exception) {
                Log.e(TAG, "Error making call", e)
            }
        }
    }

    private fun answerCall() {
        scope.launch {
            try {
                sipAudioCall?.let { call ->
                    call.answerCall(30)
                    call.startAudio()
                    
                    _currentCall.value = _currentCall.value?.copy(
                        state = CallState.CONNECTED,
                        connectTime = System.currentTimeMillis()
                    )
                    
                    enableSpeaker()
                    showCallNotification(call.peerProfile?.uriString ?: "Unknown", CallState.CONNECTED)
                }
            } catch (e: Exception) {
                Log.e(TAG, "Error answering call", e)
            }
        }
    }

                Log.d(TAG, "Call established")

#!/bin/bash

# ============================================
# VOIP Mother - Backup Script
# ============================================

set -e

# متغیرها
BACKUP_DIR="/opt/voip-backups"
DATE=$(date +%Y%m%d_%H%M%S)
PROJECT_DIR="$(dirname "$(dirname "$0")")"
RETENTION_DAYS=30

# ایجاد پوشه بکاپ
mkdir -p "$BACKUP_DIR"

print_msg() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1"
}

# بکاپ دیتابیس
backup_database() {
    print_msg "بکاپ دیتابیس..."
    
    docker exec voip-postgres pg_dump -U asterisk asterisk > "$BACKUP_DIR/db_$DATE.sql"
    gzip "$BACKUP_DIR/db_$DATE.sql"
    
    print_msg "بکاپ دیتابیس انجام شد: db_$DATE.sql.gz"
}

# بکاپ تنظیمات Asterisk
backup_asterisk_config() {
    print_msg "بکاپ تنظیمات Asterisk..."
    
    tar -czf "$BACKUP_DIR/asterisk_config_$DATE.tar.gz" \
        -C "$PROJECT_DIR/server/asterisk" .
    
    print_msg "بکاپ تنظیمات انجام شد: asterisk_config_$DATE.tar.gz"
}

# بکاپ ضبط‌ها
backup_recordings() {
    print_msg "بکاپ ضبط‌ها..."
    
    if [ -d "$PROJECT_DIR/server/recordings" ]; then
        tar -czf "$BACKUP_DIR/recordings_$DATE.tar.gz" \
            -C "$PROJECT_DIR/server" recordings
        print_msg "بکاپ ضبط‌ها انجام شد: recordings_$DATE.tar.gz"
    fi
}

# بکاپ پیام‌های صوتی
backup_voicemails() {
    print_msg "بکاپ پیام‌های صوتی..."
    
    docker cp voip-mother-server:/var/spool/asterisk/voicemail "$BACKUP_DIR/voicemail_$DATE" 2>/dev/null || true
    
    if [ -d "$BACKUP_DIR/voicemail_$DATE" ]; then
        tar -czf "$BACKUP_DIR/voicemail_$DATE.tar.gz" -C "$BACKUP_DIR" "voicemail_$DATE"
        rm -rf "$BACKUP_DIR/voicemail_$DATE"
        print_msg "بکاپ پیام‌های صوتی انجام شد: voicemail_$DATE.tar.gz"
    fi
}

# پاکسازی بکاپ‌های قدیمی
cleanup_old_backups() {
    print_msg "پاکسازی بکاپ‌های قدیمی (بیش از $RETENTION_DAYS روز)..."
    
    find "$BACKUP_DIR" -type f -mtime +$RETENTION_DAYS -delete
    
    print_msg "پاکسازی انجام شد"
}

# ایجاد لیست بکاپ‌ها
list_backups() {
    print_msg "بکاپ‌های موجود:"
    ls -lh "$BACKUP_DIR" | tail -n +2
}

# ارسال به فضای ابری (اختیاری)
upload_to_cloud() {
    if [ -n "$BACKUP_S3_BUCKET" ]; then
        print_msg "آپلود به S3..."
        aws s3 sync "$BACKUP_DIR" "s3://$BACKUP_S3_BUCKET/voip-backups/" \
            --exclude "*" \
            --include "*.gz" \
            --include "*.sql.gz"
        print_msg "آپلود به S3 انجام شد"
    fi
}

# تابع اصلی
main() {
    print_msg "شروع بکاپ‌گیری..."
    
    backup_database
    backup_asterisk_config
    backup_recordings
    backup_voicemails
    cleanup_old_backups
    list_backups
    upload_to_cloud
    
    print_msg "بکاپ‌گیری با موفقیت انجام شد"
}

import kotlinx.coroutines.launch

@AndroidEntryPoint
class MainActivity : AppCompatActivity() {

    private lateinit var binding: ActivityMainBinding
    private val viewModel: MainViewModel by viewModels()

    private val requiredPermissions = arrayOf(
        Manifest.permission.RECORD_AUDIO,
        Manifest.permission.MODIFY_AUDIO_SETTINGS,
        Manifest.permission.READ_PHONE_STATE,
        Manifest.permission.CALL_PHONE,
        Manifest.permission.USE_SIP,
        Manifest.permission.INTERNET,
        Manifest.permission.ACCESS_NETWORK_STATE,
        Manifest.permission.ACCESS_WIFI_STATE,
        Manifest.permission.CHANGE_WIFI_MULTICAST_STATE
    )

    private val permissionLauncher = registerForActivityResult(
        ActivityResultContracts.RequestMultiplePermissions()
    ) { permissions ->
        val allGranted = permissions.all { it.value }
        if (allGranted) {
            initializeServices()
        } else {
            Toast.makeText(
                this,
                "مجوزهای لازم برای اجرای VOIP ضروری است",
                Toast.LENGTH_LONG
            ).show()
        }
    }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        binding = ActivityMainBinding.inflate(layoutInflater)
        setContentView(binding.root)

        checkPermissions()
        setupNavigation()
        observeRegistrationState()
        checkUserSession()
    }

    private fun checkPermissions() {
        val permissionsToRequest = requiredPermissions.filter {
            ContextCompat.checkSelfPermission(this, it) != PackageManager.PERMISSION_GRANTED
        }.toTypedArray()

        if (permissionsToRequest.isNotEmpty()) {
            permissionLauncher.launch(permissionsToRequest)
        } else {
            initializeServices()
        }
    }

    private fun initializeServices() {
        // شروع SIP Service
        val sipServiceIntent = Intent(this, SipService::class.java).apply {
            action = SipService.ACTION_REGISTER
        }
        startService(sipServiceIntent)
    }

    private fun setupNavigation() {
        val navHostFragment = supportFragmentManager
            .findFragmentById(R.id.nav_host_fragment) as NavHostFragment
        val navController = navHostFragment.navController

        binding.bottomNavigation.setupWithNavController(navController)

        navController.addOnDestinationChangedListener { _, destination, _ ->
            when (destination.id) {
                R.id.callFragment -> hideBottomNavigation()
                R.id.dialpadFragment -> hideBottomNavigation()
                else -> showBottomNavigation()
            }
        }
    }

    private fun observeRegistrationState() {
        lifecycleScope.launch {
            viewModel.registrationState.collectLatest { state ->
                updateRegistrationUI(state)
            }
        }
    }

    private fun updateRegistrationUI(state: RegistrationState) {
        when (state) {
            is RegistrationState.Registered -> {
                binding.registrationStatus.visibility = View.GONE
            }
            is RegistrationState.Registering -> {
                binding.registrationStatus.visibility = View.VISIBLE
                binding.registrationStatus.text = "در حال اتصال..."
            }
            is RegistrationState.RegistrationFailed -> {
                binding.registrationStatus.visibility = View.VISIBLE
                binding.registrationStatus.text = "خطا در اتصال: ${state.message}"
            }
            is RegistrationState.Error -> {
                binding.registrationStatus.visibility = View.VISIBLE
                binding.registrationStatus.text = "خطا: ${state.message}"
            }
            else -> {}
        }
    }

    private fun checkUserSession() {
        val prefs = getSharedPreferences("user_prefs", MODE_PRIVATE)
        val isLoggedIn = prefs.getBoolean("is_logged_in", false)
        
        if (!isLoggedIn) {
            startActivity(Intent(this, LoginActivity::class.java))
            finish()
        }
    }

    private fun hideBottomNavigation() {
        binding.bottomNavigation.visibility = View.GONE
    }

    private fun showBottomNavigation() {
        binding.bottomNavigation.visibility = View.VISIBLE
    }

    override fun onDestroy() {
        super.onDestroy()
        // سرویس‌ها در حالت پس‌زمینه ادامه می‌یابند
    }
}

### ASSISTANT
<details type="reasoning" done="true" duration="9">
<summary>Thought for 9 seconds</summary>
> کاربر می‌خواهد همه فایل‌های پروژه را در یک فایل ZIP برای دانلود قرار دهم. متأسفانه من به عنوان یک مدل زبانی AI نمی‌توانم فایل‌ها را به صورت فیزیکی فشرده کنم یا فایل‌های باینری ایجاد کنم.
> 
> اما می‌توانم:
> 1. یک اسکریپت پایتون بنویسم که فایل‌ها را از متن تولید کند

>

> 
> بیایید این کار را انجام دهم.
</details>
متأسفانه من نمی‌توانم مستقیماً فایل ZIP تولید و ارسال کنم، اما می‌توانم یک **اسکریپت پایتون** بنویسم که تمام فایل‌های پروژه را برایتان بسازد. کافیست این اسکریپت را اجرا کنید.

def write_file(path: str, content: str):
    """نوشتن محتوا در فایل"""
    file_path = OUTPUT_DIR / path
    file_path.parent.mkdir(parents=True, exist_ok=True)
    with open(file_path, 'w', encoding='utf-8') as f:
        f.write(content)
    print(f"  ✓ {path}")

class SocketManager(
    private val serverUrl: String = "ws://192.168.1.1:8088/ws"
) {
    private val TAG = "SocketManager"
    
    private var webSocket: WebSocket? = null
    private val client = OkHttpClient.Builder()
        .readTimeout(30, TimeUnit.SECONDS)
        .writeTimeout(30, TimeUnit.SECONDS)

    fun connect() {

        }
